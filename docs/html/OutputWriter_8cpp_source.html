24 // Sort by submission ID (and reviewer ID as tiebreaker)

27 return a.reviewerId < b.reviewerId;
28 });
29
30 // Sort by reviewer ID (and submission ID as tiebreaker)
31 std::sort(byReviewer.begin(), byReviewer.end(), [](const Assignment& a, const Assignment& b) {

33 return a.submissionId < b.submissionId;
34 });
35
36 // View 1: by submission
37 file << "#SubmissionId,ReviewerId,Match" << std::endl;
38 for (const auto& a : bySubmission) {

40 }
41
42 // View 2: by reviewer
43 file << "#ReviewerId,SubmissionId,Match" << std::endl;
44 for (const auto& a : byReviewer) {
45 file << a.reviewerId << ", " << a.submissionId << ", " << a.topic << std::endl;
46 }
47
48 // Total
49 file << "#Total: " << assignments.size() << std::endl;
50
51 // Missing reviews (if any)

53 std::vector<MissingReview> sortedMissing = missingReviews;

55 return a.submissionId < b.submissionId;
56 });
57
58 file << "#SubmissionId,Domain,MissingReviews" << std::endl;
59 for (const auto& m : sortedMissing) {
60 file << m.submissionId << ", " << m.topic << ", " << m.missingReviews << std::endl;
61 }
62 }
