29bool AssignmentEngine::isCompatible(const Reviewer& reviewer, const Submission& submission) const {
30 int mode = config.getGenerateAssignments();
31
32 // Mode 0 or 1: only primary expertise matches primary domain
33 if (mode <= 1) {
34 return reviewer.getPrimaryExpertise() == submission.getPrimaryTopic();
35 }
36
37 // Mode 2: primary expertise matches primary or secondary submission domain
38 if (mode == 2) {
39 if (reviewer.getPrimaryExpertise() == submission.getPrimaryTopic()) return true;
40 if (submission.getSecondaryTopic() != -1 &&
41 reviewer.getPrimaryExpertise() == submission.getSecondaryTopic()) return true;
42 return false;
43 }
44
45 // Mode 3: any expertise matches any domain
46 if (reviewer.getPrimaryExpertise() == submission.getPrimaryTopic()) return true;
47 if (submission.getSecondaryTopic() != -1 &&
48 reviewer.getPrimaryExpertise() == submission.getSecondaryTopic()) return true;
49 if (reviewer.getSecondaryExpertise() != -1 &&
50 reviewer.getSecondaryExpertise() == submission.getPrimaryTopic()) return true;
51 if (reviewer.getSecondaryExpertise() != -1 && submission.getSecondaryTopic() != -1 &&
52 reviewer.getSecondaryExpertise() == submission.getSecondaryTopic()) return true;
53
54 return false;
55}

104void AssignmentEngine::buildBaseGraph() {
105 int numReviewers = static_cast<int>(reviewers.size());
106 int numSubmissions = static_cast<int>(submissions.size());
107
108 int maxReviewsPerReviewer = config.getMaxReviewsPerReviewer();
109
110 for (int i = 0; i < numReviewers; i++) {
111 graph.addEdge(source, getReviewerNode(i), maxReviewsPerReviewer);
112 }
113
114 for (int i = 0; i < numReviewers; i++) {
115 for (int j = 0; j < numSubmissions; j++) {
116 if (isCompatible(reviewers[i], submissions[j])) {
117 graph.addEdge(getReviewerNode(i), getSubmissionNode(j), 1);
118 }
119 }
120 }
121
122 for (int i = 0; i < numSubmissions; i++) {
123 graph.addEdge(getSubmissionNode(i), sink, config.getMinReviewsPerSubmission());
124 }
125}

127void AssignmentEngine::extractAssignments() {
128 std::vector<std::vector<Edge>>& adj = graph.getAdj();
129
130 for (int i = 0; i < static_cast<int>(reviewers.size()); i++) {
131 int reviewerNode = getReviewerNode(i);
132
133 for (const Edge& edge : adj[reviewerNode]) {
134 if (edge.to >= submissionOffset && edge.to < sink && edge.flow == 1) {
135 int submissionIndex = getSubmissionIndexFromNode(edge.to);

137 Assignment assignment;

139 assignment.submissionId = submissions[submissionIndex].getId();
140 assignment.topic = getMatchTopic(reviewers[i], submissions[submissionIndex]);
